const handleGenerate = async () => {
  if (!isLoggedIn) return notify("Please login to generate thumbnail", "error");
  if (!title.trim()) return notify("Title is required", "error");

  try {
    setLoading(true);

    const api_payload = {
      title: title,                 // top title
      topic: title,                 // center text
      additionalPrompt: additionalDetails,
      aspectRatio: aspectRatio,
      colorScheme: colorSchemaId,
      style: style,
    };

    const { data } = await api.post(
      "/api/thumbnil/generate",
      api_payload
    );

    if (data?.imageUrl) {
      setThumbnil({
        image_url: data.imageUrl,
        aspect_ratio: aspectRatio,
        title: title,
      } as IThumbnail);

      notify(data.message || "Thumbnail generated");
    } else {
      notify("Image URL not returned from server", "error");
    }
  } catch (error: any) {
    console.error("Generate error:", error);
    notify("Failed to generate thumbnail", "error");
  } finally {
    setLoading(false); // 🔥 MOST IMPORTANT LINE
  }
};
